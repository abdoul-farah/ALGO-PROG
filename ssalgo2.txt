Debut de l'algorithme 

Variables : 
	N       : la taille du tableau tab 
	tab[N]  : un tableau de taille N
	
	
Initialisation 


	

Algorithme	
	i=0
	
	Tant Que (i<N)
		Ecrire ("Veuillez intrduire une valeur")
		Lire (tab[i])
		i=i+1
	Fin Tant Que 
	
	
	

	Call TabTrier tab[N],  IsTrier

Debut du Sous-Algorithme TabTrier tab[N],  IsTrier
		ParamÃ¨tres : 
			tab[N] 
			IsTrier
			
			
		Algorithme	
		
		IsTrier=false
		j=0
		
		Tant Que j<N-1
		
			cpt1=0
			cpt2=0
			
			if tab[j]<tab[j+1]
			
				then cpt1=cpt1+1
				
			else if tab[j]>tab[j+1]
				then cpt2=cpt2+1
				
			end if 
			
			j=j+1
		
		Fin Tant Que 
		
		if cpt1==N-1 OU cpt2=N-1
			then IsTrier=true
		else
			IsTrier=false
		
Fin du Sous-Algorithme
